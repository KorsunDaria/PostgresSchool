static Node *simplify_bool_expr(BoolExpr *expr);


static Node *
simplify_bool_expr(BoolExpr *expr)
{
	List	   *newargs = NIL;
	ListCell   *lc;

	Assert(expr->boolop == AND_EXPR || expr->boolop == OR_EXPR);

	foreach(lc, expr->args)
	{
		Node	   *arg = (Node *) lfirst(lc);

		if (IsA(arg, Const))
		{
			Const	   *c = (Const *) arg;

			
			if (c->consttype == BOOLOID && !c->constisnull)
			{
				bool		val = DatumGetBool(c->constvalue);

				if ((expr->boolop == AND_EXPR && val == false) ||
					(expr->boolop == OR_EXPR && val == true))
					return (Node *) makeBoolConst(val, false);
				continue;
			}
		}

		newargs = lappend(newargs, arg);
	}

	if (newargs == NIL)
		return (Node *) makeBoolConst(expr->boolop == AND_EXPR, false);

	if (list_length(newargs) == 1)
		return (Node *) linitial(newargs);

	expr->args = newargs;
	return (Node *) expr;
}

static Node *
simplify_bool_mutator(Node *node, void *context)
{
	if (node == NULL)
		return NULL;

	if (IsA(node, BoolExpr))
	{
		BoolExpr   *expr = (BoolExpr *) node;

		if (expr->boolop == AND_EXPR || expr->boolop == OR_EXPR)
		{
			BoolExpr   *copy = makeNode(BoolExpr);
			ListCell   *lc;
			List	   *newargs = NIL;

			copy->boolop = expr->boolop;
			copy->location = expr->location;

			
			foreach(lc, expr->args)
			{
				Node	   *newarg = simplify_bool_mutator((Node *) lfirst(lc),
															context);

				newargs = lappend(newargs, newarg);
			}
			copy->args = newargs;

			return simplify_bool_expr(copy);
		}
	}

	return expression_tree_mutator(node, simplify_bool_mutator, context);
}

void
simplify_query_quals(Query *query)
{
	if (query->jointree != NULL)
		query->jointree = (FromExpr *)
			simplify_bool_mutator((Node *) query->jointree, NULL);

	if (query->havingQual != NULL)
		query->havingQual =
			simplify_bool_mutator(query->havingQual, NULL);
}
