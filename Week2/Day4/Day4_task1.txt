Настроть потоковую реплкацю

initdb -D ./primary -U postgres --locale=C --encoding=UTF8

listen_addresses = '*'
port = 5432
wal_level = replica
max_wal_senders = 10
max_replication_slots = 10
hot_standby = on

./primary/pg_hba.conf::host replication postgres 127.0.0.1/32 trust

pg_basebackup -h 127.0.0.1 -p 5432 -U postgres \
  -D ./standby1 -P -v -R -X stream \
  -S standby1_slot -C

./standby1/postgresql.conf::

port = 5433
hot_standby = on

psql -U postgres -d postgres -p 5432
psql -U postgres -d postgres -p 5433

create table t(id int); insert into t values (1);
select * from t

autovacuum_naptime = 5s 

/Users/dariakorsun/pg/standby1/postgresql.conf
hot_standby_feedback = off

psql -U postgres -h 127.0.0.1 -p 5432 -c "drop table if exists t2; create table t2(id int primary key, data text); 
insert into t2 values (1, 'old');"

psql -U postgres -h 127.0.0.1 -p 5433
begin transaction isolation level repeatable read;
